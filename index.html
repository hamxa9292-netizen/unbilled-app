// =================================================== // UNBILLED READINGS — Google Apps Script API // =================================================== const SHEET_NAME = "unbilled"; const USERS_SHEET = "users"; // Sheet with login credentials const DATA_START_ROW = 2; const COL_ACNO = 2; const COL_ADDRESS = 5; const COL_METER = 4; const COL_AREA = 7; const COL_READING = 8; const COL_PREV_DATE = 9; const COL_RDATE = 12; const COL_NREADING = 13; const COL_NEWDATE = 14; const COL_REMARKS = 21; const COL_PHOTO = 22; // Users sheet columns: A=username, B=password, C=name, D=area const U_USERNAME = 1; const U_PASSWORD = 2; const U_NAME = 3; const U_AREA = 4; function doGet(e) { const action = e.parameter.action; const acno = e.parameter.acno; const area = e.parameter.area; if (action === "login") return doLogin(e.parameter.username, e.parameter.password); if (action === "search" && acno) return searchAccount(acno); if (action === "all") return getAllAccounts(area); if (action === "areas") return getAreas(); if (action === "pending") return getPendingAccounts(area); return jsonResponse({ error: "Invalid request" }); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === "update") return updateReading(data); if (data.action === "upload_photo") return uploadPhoto(data); if (data.action === "save_photo_url") return savePhotoUrl(data); return jsonResponse({ error: "Invalid action" }); } catch (err) { return jsonResponse({ error: err.message }); } } // ---- LOGIN ---- function doLogin(username, password) { if (!username || !password) return jsonResponse({ success: false, message: "Missing credentials" }); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(USERS_SHEET); // Auto-create users sheet if missing with sample data if (!sheet) { sheet = ss.insertSheet(USERS_SHEET); sheet.getRange(1,1,1,4).setValues([["Username","Password","Name","Area"]]); sheet.getRange(2,1,1,4).setValues([["hamza","1234","Hamza","11"]]); sheet.getRange(3,1,1,4).setValues([["azim","1234","Azim","12"]]); } const lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ success: false, message: "No users found" }); const values = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (let i = 0; i < values.length; i++) { const row = values[i]; if (String(row[U_USERNAME-1]).trim().toLowerCase() === username.toLowerCase() && String(row[U_PASSWORD-1]).trim() === password) { return jsonResponse({ success: true, user: { username: String(row[U_USERNAME-1]).trim(), name: String(row[U_NAME-1]).trim(), area: String(row[U_AREA-1]).trim() } }); } } return jsonResponse({ success: false, message: "Invalid username or password" }); } // ---- Save photo URL ---- function savePhotoUrl(data) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const lastRow = sheet.getLastRow(); for (let r = DATA_START_ROW; r <= lastRow; r++) { if (String(sheet.getRange(r, COL_ACNO).getValue()).trim() === String(data.acno).trim()) { sheet.getRange(r, COL_PHOTO).setFormula('=HYPERLINK("' + data.photoUrl + '","📷 View Photo")'); logUpdate(data.acno, data.worker||'Field', '', '', 'Photo: ' + data.photoUrl); return jsonResponse({ success: true }); } } return jsonResponse({ success: false, error: "Account not found" }); } catch(err) { return jsonResponse({ success: false, error: err.message }); } } // ---- Get all unique areas ---- function getAreas() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) return jsonResponse({ error: "Sheet not found" }); const lastRow = sheet.getLastRow(); const values = sheet.getRange(DATA_START_ROW, COL_AREA, lastRow - DATA_START_ROW + 1, 1).getValues(); const areas = new Set(); values.forEach(r => { if (r[0]) areas.add(String(r[0]).trim()); }); return jsonResponse({ areas: Array.from(areas).sort((a,b) => Number(a)-Number(b)) }); } // ---- Get all accounts ---- function getAllAccounts(area) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) return jsonResponse({ error: "Sheet not found" }); const lastRow = sheet.getLastRow(); if (lastRow < DATA_START_ROW) return jsonResponse({ accounts: [], total: 0 }); const range = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, COL_PHOTO); const values = range.getValues(); const accounts = []; for (let i = 0; i < values.length; i++) { const row = values[i]; const acno = String(row[COL_ACNO - 1]).trim(); if (!acno || acno === '') continue; const rowArea = String(row[COL_AREA - 1]).trim(); if (area && area !== 'all' && rowArea !== String(area).trim()) continue; const rdate = row[COL_RDATE - 1]; const prevDate = row[COL_PREV_DATE - 1]; const newDate = row[COL_NEWDATE - 1]; accounts.push({ row: DATA_START_ROW + i, acno: acno, area: rowArea, address: row[COL_ADDRESS - 1] || '', meter: String(row[COL_METER - 1] || ''), prevReading: row[COL_READING - 1] || '', prevDate: prevDate ? Utilities.formatDate(new Date(prevDate), "GMT+5", "dd/MM/yyyy") : '', rdate: rdate ? Utilities.formatDate(new Date(rdate), "GMT+5", "dd/MM/yyyy") : '', newReading: row[COL_NREADING - 1] || '', newDate: newDate ? Utilities.formatDate(new Date(newDate), "GMT+5", "dd/MM/yyyy") : '', remarks: row[COL_REMARKS - 1] || '', photoUrl: row[COL_PHOTO - 1] || '', status: newDate ? 'done' : 'pending' }); } return jsonResponse({ accounts: accounts, total: accounts.length }); } function searchAccount(acno) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) return jsonResponse({ found: false }); const lastRow = sheet.getLastRow(); const values = sheet.getRange(DATA_START_ROW, 1, lastRow - DATA_START_ROW + 1, COL_PHOTO).getValues(); for (let i = 0; i < values.length; i++) { const row = values[i]; if (String(row[COL_ACNO-1]).trim() !== String(acno).trim()) continue; const rdate = row[COL_RDATE-1], prevDate = row[COL_PREV_DATE-1], newDate = row[COL_NEWDATE-1]; return jsonResponse({ found: true, row: DATA_START_ROW + i, acno: String(row[COL_ACNO-1]).trim(), area: String(row[COL_AREA-1]||''), address: row[COL_ADDRESS-1]||'', meter: String(row[COL_METER-1]||''), prevReading: row[COL_READING-1]||'', prevDate: prevDate ? Utilities.formatDate(new Date(prevDate),"GMT+5","dd/MM/yyyy") : '', rdate: rdate ? Utilities.formatDate(new Date(rdate),"GMT+5","dd/MM/yyyy") : '', newReading: row[COL_NREADING-1]||'', newDate: newDate ? Utilities.formatDate(new Date(newDate),"GMT+5","dd/MM/yyyy") : '', remarks: row[COL_REMARKS-1]||'', photoUrl: row[COL_PHOTO-1]||'', status: newDate ? 'done' : 'pending' }); } return jsonResponse({ found: false, message: "Account not found" }); } function updateReading(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const row = parseInt(data.row); if (!row || row < DATA_START_ROW) return jsonResponse({ success: false, error: "Invalid row" }); const rowAcno = String(sheet.getRange(row, COL_ACNO).getValue()).trim(); if (rowAcno !== String(data.acno).trim()) return jsonResponse({ success: false, error: "Account mismatch" }); if (data.newReading !== undefined && data.newReading !== '') sheet.getRange(row, COL_NREADING).setValue(Number(data.newReading)); if (data.newDate && data.newDate !== '') { const parts = data.newDate.split('-'); sheet.getRange(row, COL_NEWDATE).setValue(new Date(parts[0], parts[1]-1, parts[2])); sheet.getRange(row, COL_NEWDATE).setNumberFormat("DD/MM/YYYY"); } if (data.remarks !== undefined) sheet.getRange(row, COL_REMARKS).setValue(data.remarks); logUpdate(data.acno, data.worker||'Field', data.newReading, data.newDate, data.remarks); return jsonResponse({ success: true, message: "Updated ACNO " + data.acno }); } function uploadPhoto(data) { try { let folder; const folders = DriveApp.getFoldersByName("UnbilledPhotos"); folder = folders.hasNext() ? folders.next() : DriveApp.createFolder("UnbilledPhotos"); const imageData = Utilities.base64Decode(data.photo); const blob = Utilities.newBlob(imageData, "image/jpeg", "ACNO_" + data.acno + "_" + (data.date || new Date().toISOString().split('T')[0]) + ".jpg"); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const photoUrl = "https://drive.google.com/uc?export=view&id=" + file.getId(); const viewUrl = "https://drive.google.com/file/d/" + file.getId() + "/view"; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const lastRow = sheet.getLastRow(); for (let r = DATA_START_ROW; r <= lastRow; r++) { if (String(sheet.getRange(r, COL_ACNO).getValue()).trim() === String(data.acno).trim()) { sheet.getRange(r, COL_PHOTO).setFormula('=HYPERLINK("' + viewUrl + '","📷 View Photo")'); break; } } logUpdate(data.acno, data.worker||'Field', '', data.date, 'Photo uploaded: ' + viewUrl); return jsonResponse({ success: true, url: photoUrl }); } catch (err) { return jsonResponse({ success: false, error: err.message }); } } function getPendingAccounts(area) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const lastRow = sheet.getLastRow(); const values = sheet.getRange(DATA_START_ROW, 1, lastRow-DATA_START_ROW+1, COL_NEWDATE).getValues(); const pending = []; for (let i = 0; i < values.length; i++) { if (!values[i][COL_NEWDATE-1] && values[i][COL_ACNO-1]) { const rowArea = String(values[i][COL_AREA-1]||'').trim(); if (area && area !== 'all' && rowArea !== String(area).trim()) continue; pending.push({ acno: String(values[i][COL_ACNO-1]), area: rowArea, address: values[i][COL_ADDRESS-1]||'' }); } } return jsonResponse({ count: pending.length, accounts: pending }); } function logUpdate(acno, worker, reading, date, remarks) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let log = ss.getSheetByName("_UpdateLog"); if (!log) { log = ss.insertSheet("_UpdateLog"); log.getRange(1,1,1,6).setValues([["Timestamp","ACNO","Worker","New Reading","New Date","Remarks"]]); } log.appendRow([new Date(), acno, worker, reading, date, remarks]); } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
